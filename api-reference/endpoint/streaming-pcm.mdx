## Fixed Headers

<ParamField header="Accept" type="audio/pcm" required></ParamField>

## Variable Parameters

<ParamField body="speaker" type="string" required>
  Must be one of the voices <a href="https://rime.ai/docs/voices">listed in our documentation</a>.
</ParamField>
<ParamField body="text" type="string" required>
  The text you'd like spoken
</ParamField>

<ParamField body="modelId" type="string">
  Choose `mist` for hyper-realistic conversational voices or `v1` for Rime's first-gen model (default: `v1`)
</ParamField>

<ParamField body="samplingRate" type="int">
  The value, if provided, must be between 4000 and 44100. Default: 16000
</ParamField>

<ParamField body="speedAlpha" type="float" default="1.0">
  Adjusts the speed of speech. Lower than 1.0 is faster than default. Higher than 1.0 is slower than default.
</ParamField>

<ParamField body="reduceLatency" type="bool" default="false">
  Reduces the latency of response, at the cost of some possible mispronunciation of digits and abbreviations.
</ParamField>

```go Go
package main

import (
  "fmt"
  "strings"
  "net/http"
  "io/ioutil"
)

func main() {

  url := "https://users.rime.ai/v1/rime-tts"

  payload := strings.NewReader("{\n  \"speaker\": \"<string>\",\n  \"text\": \"<string>\",\n  \"modelId\": \"<string>\",\n  \"samplingRate\": 16000,\n  \"speedAlpha\": 1.0,\n  \"reduceLatency\": false\n}")

  req, _ := http.NewRequest("POST", url, payload)

  req.Header.Add("Accept", "audio/pcm")
  req.Header.Add("Authorization", "Bearer <authorization>")
  req.Header.Add("Content-Type", "application/json")

  res, _ := http.DefaultClient.Do(req)

  defer res.Body.Close()
  body, _ := ioutil.ReadAll(res.Body)

  fmt.Println(res)
  fmt.Println(string(body))

}
```
