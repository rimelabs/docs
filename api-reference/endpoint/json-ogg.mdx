## Fixed Parameters

<ParamField body="audioFormat" type="ogg" required></ParamField>

## Variable Parameters

<ParamField body="speaker" type="string" required>
  Must be the lowercase spelling of one of the voices <a href="https://rime.ai/docs/voices">listed in our documentation</a>.
</ParamField>

<ParamField body="text" type="string" required>
  The text you'd like spoken
</ParamField>

<ParamField body="samplingRate" type="int" required>
  Must be one of `8000`, `12000`, `16000`, or `24000`
</ParamField>

<ParamField body="speedAlpha" type="float" default="1.0">
  Adjusts the speed of speech. Lower than 1.0 is faster than default. Higher than 1.0 is slower than default.
</ParamField>

<ParamField body="reduceLatency" type="bool" default="false">
  Reduces the latency of response, at the cost of some possible mispronunciation of digits and abbreviations.
</ParamField>

```go Go
package main

import (
  "fmt"
  "strings"
  "net/http"
  "io/ioutil"
)

func main() {

  url := "http://fake.stvs.tv/audio"

  payload := strings.NewReader("{\n  \"speaker\": \"<string>\",\n  \"text\": \"<string>\",\n  \"audioFormat\": \"ogg\",\n  \"samplingRate\": 22050,\n  \"speedAlpha\": 1.0,\n  \"reduceLatency\": false\n}")

  req, _ := http.NewRequest("POST", url, payload)

  req.Header.Add("Accept", "application/json")
  req.Header.Add("Authorization", "Bearer <authorization>")
  req.Header.Add("Content-Type", "application/json")

  res, _ := http.DefaultClient.Do(req)

  defer res.Body.Close()
  body, _ := ioutil.ReadAll(res.Body)

  fmt.Println(res)
  fmt.Println(string(body))

}
```
